使用Hifiasem利用PacBio HiFi数据进行基因组组装

mkdir -p /home/train/04.genome_assembling/Hifiasm
cd /home/train/04.genome_assembling/Hifiasm
/opt/biosoft/miniconda3_for_pb-assembly/bin/ccs ~/00.incipient_data/data_for_genome_assembling/lambdaTINY_Sequel2/m54026_181219_010936_tiny.subreads.bam ccs.bam
/opt/biosoft/miniconda3_for_pb-assembly/bin/bam2fasta -o ccs_reads -u ccs.bam

# 使用hifiasm命令进行基因组组装
# -o 设置输出文件前缀。默认值为hifiasm.asm。最终结果文件后缀为.bp.p_ctg.gfa。
# -t 设置运行线程数。默认值为1。
# -l 参数设置purge水平。0表示不消除重复序列；1轻微消除；2/3则猛烈消除。当对纯合或近交样品基因组进行组装时，使用参数值0；当需将来自父本和母本的两套基因组序列同时得到并分开时，使用参数值0；默认设置下，使用-1和-2参数输入双亲Illumina数据时，该参数值为0，否则默认为3，表示Hifiasm用于组装得到单倍基因组序列。
hifiasm -o hifiasm.asm -t 4 -l 0 ccs_reads.fasta
perl -e 'while (<>) { if (m/^S/) { @_ = split /\t/; print ">$_[1]\n$_[2]\n"; } }' hifiasm.asm.bp.p_ctg.gfa > hifiasm.asm.fasta

使用NextPolish进行基因组序列校正

mkdir -p /home/train/04.genome_assembling/NextPolish
cd /home/train/04.genome_assembling/NextPolish
ln -s ~/03.sequencing_data_quality_control/FastUniq/illumina.?.fastq ./
ln -s ../purge_dups/subreads.fasta ./
ls $PWD/*.fastq > sgs.fofn
ls $PWD/*.fasta > lgs.fofn
ln -s ~/00.incipient_data/data_for_genome_assembling/assemblies_of_Malassezia_sympodialis/purge_dups.fasta raw.genome.fasta
cat << EOF > run.cfg
[General]
job_type = local
# 设置任务类型：设置为local表示使用单台计算机进行计算；设置为为sge会使用集群进行计算。
job_prefix = nextPolish
task = best
# 设置任务流程：1和2表示使用short reads进行修正的算法；3和4也是short reads进行修正的算法，实验性操作，不推荐使用；5表示使用long reads进行修正算法。比如该值为555512121212则是先用long reads修正迭代4次，再使用short reads的12迭代4次。设置为best，则等于551212，先使用long reads迭代2次，再用12迭代2次，是推荐的值。
rewrite = no
# 是否覆盖已有的文件夹。默认设置下，程序中断后，可以继续接着运行。
rerun = 3
# 重头运行未完成的任务，直到任务结束或失败3次。
parallel_jobs = 1
# 设置并行化的任务数。程序会将高通量测序数据分割成相应的份数。
multithread_jobs = 8
# 设置程序调用bwa或minimap2进行比对时所使用的线程数。
genome = ./raw.genome.fasta
genome_size = auto
workdir = ./01_rundir
polish_options = -p 8
# 设置程序调用NextPolish/lib/nextpolish2.py对基因组进行修正时所使用的线程数。

[sgs_option]
sgs_fofn = ./sgs.fofn
sgs_options = -max_depth 100 -bwa
# 最多使用100x的short reads数据进行修正。

[lgs_option]
lgs_fofn = ./lgs.fofn
lgs_options = -min_read_len 1k -max_depth 100
# 最多使用100x的long reads数据进行修正。
lgs_minimap2_options = -x map-pb -t 4
# 使用pacbio数据进行修正。
EOF
#export DRMAA_LIBRARY_PATH=/opt/sysoft/sge/lib/lx-amd64/libdrmaa.so.1.0
nextPolish run.cfg
genome_seq_clear.pl --seq_prefix nextpolish 01_rundir/genome.nextpolish.fasta > nextpolish.fasta
